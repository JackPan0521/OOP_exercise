import java.util.ArrayList;
import java.util.List;

class WeatherStation implements ISubject{
    private List obsList = new ArrayList<>();
    private String weather;
    
    public void setWeather(String weather) {
        this.weather = weather;
        notifyDisplays();
    }
    
    public String getWeather(){
    	return weather;
    }
    
    @override
    public void registerObserver(IObserver obs){
    	obsList.add(obs);
    }
    
    @override
    public void removeObserver(IObserver obs){
    	obsList.remove(obs);
    }
    
    @override
    public void notifyDisplays() {
        for(IObserver o:obsList){
        	o.update(this.getClass().getSimpleName(),getWeather());
        }
    }
}

	void registerObserver(IObserver obs);
    void removeObserver(IObserver obs);
    void notifyDisplays();
}

	void update();

新設計有何優點?
本設計以Observer Pattern為核心，改善原本氣象站（WeatherStation）與顯示單位（WeatherDisplay）高度耦合的問題。舊有設計中，氣象站直接推送資料，當觀察者數量或型態改變時，需修改內部邏輯，維護困難。
改良後透過 IObserver 與 ISubject 介面分離角色，並實作註冊（Register）、移除（Remove）與通知（Notify）機制，使氣象站僅負責管理觀察者清單與發送更新。update 方法加入資料來源名稱參數，提升系統彈性，使觀察者可同時監控多個來源。
此設計具備低耦合、高擴充性與良好維護性，符合開放封閉原則（OCP），未來新增不同顯示裝置或資料來源時，無需修改既有程式架構，大幅提升系統穩定性與延展性。!@#$
