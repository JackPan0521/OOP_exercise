import java.util.ArrayList;
import java.util.List;

class PriceTracker implements ISubject{
    private List shopperList = new ArrayList<>();
    private int price;

    @Override
    public void setPrice(int price){
        this.price = price;
        notifyObservers();
    }
    
    @Override
    public void removeObserver(IObserver obs){
        shopperList.remove(obs);
    }
    
    @Override
    public void registerObserver(IObserver obs){
        shopperList.add(obs);
    }

    private void notifyObservers(){
        for (IObserver o : shopperList) {
            o.update(price);
        }
    }
}
class Shopper implements IObserver{
    private String shopName;

    public Shopper(String shopName){
        this.shopName = shopName;
    }

    @Override
    public void update(int price){
        System.out.println(this.shopName + " received update from PriceTracker. New Watermelon price:" + price);
    }

    public String getShopName(){
        return shopName;
    }
}

interface IObserver {
    void update(int price);
}

interface ISubject {
    void registerObserver(IObserver obs);
    void removeObserver(IObserver obs);
    void setPrice(int price);
}
