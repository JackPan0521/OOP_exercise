package MidExam115.Q3;

import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

public class ConnectionPool {
    private List pool;
    private Set borrowed;

    public ConnectionPool(int size) {
        pool = new ArrayList<>();
        borrowed = new HashSet<>();
        for (int i = 0; i < size; i++) {
            pool.add(new Connection());
        }
    }

    public Connection borrowConnection() {
        if (pool.isEmpty()) {
            System.out.println("No available connections.");
            return null;
        }
        Connection conn = pool.remove(pool.size() - 1);
        borrowed.add(conn);
        return conn;
    }

    public void returnConnection(Connection conn) {
        if (!borrowed.contains(conn)) {
            System.out.println("Warning: This connection was not borrowed from the pool.");
            return;
        }
        borrowed.remove(conn);
        pool.add(conn);
    }
}

透過 `ConnectionPool` 預先建立固定數量的 `Connection` 物件並統一管理。`borrowConnection()` 從池中取出可用連線並記錄至 `borrowed` 集合；`returnConnection()` 歸還時會先驗證該連線是否確實來自池中，防止重複歸還或非法歸還。此設計避免頻繁建立與銷毀物件的效能消耗，適用於資料庫連線等建立成本高的資源管理場景，符合**資源重用**原則，有效提升系統效能與穩定性。!@#$
